MySQL 使用手册

1. MySQL 基础概念

1.1 什么是 MySQL

MySQL 是一个开源的关系型数据库管理系统（RDBMS），使用 SQL 作为查询语言。具有体积小、速度快、成本低、开源等特点，是互联网应用中最常用的数据库。

1.2 MySQL 的特点

  • 开源免费
  • 跨平台
  • 可靠性强
  • 支持事务处理
  • 查询速度快
  • 安全性高

2. MySQL 安装与配置

2.1 Windows 安装

  1. 下载 MySQL 安装程序
  2. 运行安装向导
  3. 选择安装类型（开发者默认、服务器）
  4. 配置 MySQL 服务端口和字符集
  5. 初始化数据库和创建管理员账户

2.2 Linux 安装

# Ubuntu/Debian
sudo apt-get install mysql-server

# CentOS/RHEL
sudo yum install mysql-server

# 启动 MySQL 服务
sudo systemctl start mysql
sudo systemctl enable mysql

2.3 macOS 安装

# 使用 Homebrew
brew install mysql
brew services start mysql

2.4 配置文件

MySQL 配置文件通常位于 /etc/mysql/my.cnf（Linux）或 my.ini（Windows）

关键配置项：

  • port: 数据库端口（默认 3306）
  • max_connections: 最大连接数
  • character_set_server: 服务器字符集
  • datadir: 数据存储目录

3. 数据库基本操作

3.1 连接 MySQL

mysql -u root -p
# 或指定主机和端口
mysql -h localhost -P 3306 -u root -p

3.2 数据库操作

-- 查看所有数据库
SHOW DATABASES;

-- 创建数据库
CREATE DATABASE IF NOT EXISTS mydb CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 选择数据库
USE mydb;

-- 删除数据库
DROP DATABASE IF EXISTS mydb;

-- 查看数据库信息
SHOW CREATE DATABASE mydb;

3.3 表的基本操作

-- 创建表
CREATE TABLE users (
  id INT PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(50) NOT NULL UNIQUE,
  email VARCHAR(100) NOT NULL,
  age INT DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- 查看表结构
DESC users;
SHOW CREATE TABLE users;

-- 修改表
ALTER TABLE users ADD COLUMN phone VARCHAR(20);
ALTER TABLE users MODIFY COLUMN age INT DEFAULT 18;
ALTER TABLE users DROP COLUMN phone;

-- 删除表
DROP TABLE IF EXISTS users;

-- 重命名表
RENAME TABLE users TO customers;

5. SQL 增删改查

5.1 INSERT 插入数据

-- 插入单条记录
INSERT INTO users (username, email, age) VALUES ('john', 'john@example.com', 25);

-- 插入多条记录
INSERT INTO users (username, email, age) VALUES 
  ('jane', 'jane@example.com', 28),
  ('bob', 'bob@example.com', 30);

-- 从另一个表插入
INSERT INTO users (username, email, age) 
SELECT name, email, age FROM tmp_users;

5.2 SELECT 查询数据

-- 基本查询
SELECT * FROM users;
SELECT username, email FROM users;

-- 条件查询
SELECT * FROM users WHERE age > 25;
SELECT * FROM users WHERE username = 'john' AND age > 20;
SELECT * FROM users WHERE age BETWEEN 20 AND 30;
SELECT * FROM users WHERE username IN ('john', 'jane', 'bob');
SELECT * FROM users WHERE email LIKE '%@example.com';

-- 排序
SELECT * FROM users ORDER BY age ASC;
SELECT * FROM users ORDER BY created_at DESC LIMIT 10;

-- 分页
SELECT * FROM users LIMIT 10 OFFSET 20;

-- 聚合函数
SELECT COUNT(*) FROM users;
SELECT SUM(age), AVG(age), MAX(age), MIN(age) FROM users;

-- 分组
SELECT age, COUNT(*) FROM users GROUP BY age;
SELECT age, COUNT(*) FROM users GROUP BY age HAVING COUNT(*) > 1;

5.3 UPDATE 更新数据

-- 更新单个字段
UPDATE users SET age = 26 WHERE username = 'john';

-- 更新多个字段
UPDATE users SET age = 30, email = 'newemail@example.com' WHERE id = 1;

-- 条件更新
UPDATE users SET age = age + 1 WHERE age < 25;

-- 更新所有记录
UPDATE users SET updated_at = NOW();

5.4 DELETE 删除数据

-- 删除指定记录
DELETE FROM users WHERE id = 1;

-- 条件删除
DELETE FROM users WHERE age < 20;

-- 删除所有数据（保留表结构）
DELETE FROM users;

-- 清空表（效率更高）
TRUNCATE TABLE users;

6. 索引

6.1 索引类型

  • PRIMARY KEY: 主键索引，唯一且非空
  • UNIQUE: 唯一索引，唯一但可为空
  • INDEX: 普通索引
  • FULLTEXT: 全文索引
  • COMPOSITE: 复合索引

6.2 索引操作

-- 创建索引
CREATE INDEX idx_username ON users(username);
CREATE UNIQUE INDEX idx_email ON users(email);
CREATE INDEX idx_age_email ON users(age, email);

-- 查看索引
SHOW INDEX FROM users;

-- 删除索引
DROP INDEX idx_username ON users;

-- 在 ALTER TABLE 中创建索引
ALTER TABLE users ADD INDEX idx_age (age);
ALTER TABLE users ADD UNIQUE INDEX idx_email (email);

6.3 索引最佳实践

  • 在经常用于 WHERE、ORDER BY、GROUP BY 的列上创建索引
  • 避免在低基数列上创建索引
  • 定期分析和优化索引
  • 避免过多索引影响写入性能

7. 关键字与约束

7.1 常用约束

CREATE TABLE products (
  id INT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(100) NOT NULL,
  price DECIMAL(10, 2) NOT NULL,
  stock INT DEFAULT 0,
  category_id INT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY uk_name (name),
  FOREIGN KEY (category_id) REFERENCES categories(id)
);

7.2 约束说明

  • NOT NULL: 非空约束
  • PRIMARY KEY: 主键，唯一且非空
  • UNIQUE: 唯一约束
  • DEFAULT: 默认值
  • AUTO_INCREMENT: 自动递增
  • FOREIGN KEY: 外键约束
  • CHECK: 检查约束

8. 连接与关联

8.1 内连接 (INNER JOIN)

SELECT u.username, o.order_id 
FROM users u 
INNER JOIN orders o ON u.id = o.user_id;

8.2 左连接 (LEFT JOIN)

SELECT u.username, COUNT(o.id) 
FROM users u 
LEFT JOIN orders o ON u.id = o.user_id 
GROUP BY u.id;

8.3 右连接 (RIGHT JOIN)

SELECT u.username, o.order_id 
FROM users u 
RIGHT JOIN orders o ON u.id = o.user_id;

8.4 交叉连接 (CROSS JOIN)

SELECT * FROM users CROSS JOIN categories;

8.5 自连接

SELECT a.id, a.name, b.name as parent_name
FROM categories a 
LEFT JOIN categories b ON a.parent_id = b.id;

9. 事务处理

9.1 事务特性（ACID）

  • 原子性 (Atomicity): 事务要么全部执行，要么全部回滚
  • 一致性 (Consistency): 事务前后数据一致
  • 隔离性 (Isolation): 事务互不影响
  • 持久性 (Durability): 提交后持久保存

9.2 事务操作

-- 开始事务
START TRANSACTION;
-- 或
BEGIN;

-- 执行 SQL 操作
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
UPDATE accounts SET balance = balance + 100 WHERE id = 2;

-- 提交事务
COMMIT;

-- 回滚事务
ROLLBACK;

-- 设置保存点
SAVEPOINT sp1;
-- 回滚到保存点
ROLLBACK TO sp1;

9.3 隔离级别

-- 查看隔离级别
SELECT @@transaction_isolation;

-- 设置隔离级别
SET TRANSACTION ISOLATION LEVEL READ UNCOMMITTED;
SET TRANSACTION ISOLATION LEVEL READ COMMITTED;
SET TRANSACTION ISOLATION LEVEL REPEATABLE READ;
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;

10. 性能优化

10.1 查询优化

  • 使用 EXPLAIN 分析查询计划
  • 避免 SELECT *，只查询需要的列
  • 使用索引加速查询
  • 避免在 WHERE 中使用函数
  • 使用连接代替子查询

10.2 EXPLAIN 查询分析

EXPLAIN SELECT * FROM users WHERE age > 25;

-- 关键字段说明：
-- id: 查询序列号
-- select_type: 查询类型（SIMPLE、PRIMARY、SUBQUERY 等）
-- type: 访问类型（ALL、index、range、ref、eq_ref、const 等）
-- key: 使用的索引
-- rows: 扫描行数
-- Extra: 额外信息

10.3 常见优化建议

-- 不好的做法
SELECT * FROM users WHERE YEAR(created_at) = 2024;

-- 优化后
SELECT * FROM users WHERE created_at >= '2024-01-01' AND created_at < '2025-01-01';

-- 不好的做法
SELECT * FROM orders WHERE amount/100 > 50;

-- 优化后
SELECT * FROM orders WHERE amount > 5000;

11. 用户与权限管理

11.1 用户管理

-- 创建用户
CREATE USER 'app_user'@'localhost' IDENTIFIED BY 'password123';
CREATE USER 'app_user'@'%' IDENTIFIED BY 'password123';  -- 任意主机

-- 修改密码
ALTER USER 'app_user'@'localhost' IDENTIFIED BY 'newpassword';

-- 删除用户
DROP USER 'app_user'@'localhost';

-- 查看用户
SELECT * FROM mysql.user;

11.2 权限管理

-- 授予权限
GRANT SELECT, INSERT, UPDATE, DELETE ON mydb.* TO 'app_user'@'localhost';
GRANT ALL PRIVILEGES ON mydb.* TO 'app_user'@'localhost';
GRANT SELECT ON mydb.users TO 'read_user'@'localhost';

-- 撤销权限
REVOKE SELECT, INSERT ON mydb.* FROM 'app_user'@'localhost';

-- 查看权限
SHOW GRANTS FOR 'app_user'@'localhost';

-- 刷新权限
FLUSH PRIVILEGES;

12. 备份与恢复

12.1 使用 mysqldump 备份

# 备份单个数据库
mysqldump -u root -p mydb > mydb_backup.sql

# 备份多个数据库
mysqldump -u root -p --databases db1 db2 > backup.sql

# 备份所有数据库
mysqldump -u root -p --all-databases > all_backup.sql

# 备份特定表
mysqldump -u root -p mydb users products > tables_backup.sql

12.2 恢复数据库

# 恢复单个数据库
mysql -u root -p mydb < mydb_backup.sql

# 恢复所有数据库
mysql -u root -p < all_backup.sql

# 从 SQL 文件恢复
mysql> source /path/to/backup.sql;

12.3 增量备份

# 启用二进制日志（在 my.cnf 中设置）
log-bin=mysql-bin

# 查看二进制日志
SHOW BINARY LOGS;

# 恢复特定时间点
mysqlbinlog --start-datetime="2024-01-01 10:00:00" \
            --stop-datetime="2024-01-01 12:00:00" \
            mysql-bin.000001 | mysql -u root -p

13. 常用函数

13.1 字符串函数

SELECT 
  CONCAT('Hello', ' ', 'World'),       -- 字符串拼接
  LENGTH('hello'),                      -- 字符串长度
  UPPER('hello'),                       -- 转大写
  LOWER('HELLO'),                       -- 转小写
  SUBSTR('hello', 1, 3),               -- 字符串截取
  TRIM('  hello  '),                   -- 去除空格
  REPLACE('hello world', 'world', 'MySQL');  -- 字符串替换

13.2 数值函数

SELECT 
  ROUND(3.14159, 2),      -- 四舍五入
  CEIL(3.14),             -- 向上取整
  FLOOR(3.99),            -- 向下取整
  ABS(-10),               -- 绝对值
  RAND(),                 -- 随机数
  POWER(2, 3),            -- 幂运算
  SQRT(16);               -- 平方根

13.3 日期函数

SELECT 
  NOW(),                          -- 当前日期时间
  CURDATE(),                      -- 当前日期
  CURTIME(),                      -- 当前时间
  DATE_ADD(NOW(), INTERVAL 1 DAY),    -- 日期加法
  DATEDIFF('2024-12-31', '2024-01-01'),  -- 日期差
  DATE_FORMAT(NOW(), '%Y-%m-%d'),    -- 日期格式化
  YEAR(NOW()),                   -- 提取年份
  MONTH(NOW()),                  -- 提取月份
  DAY(NOW());                    -- 提取日期

14. 视图

14.1 创建视图

-- 创建视图
CREATE VIEW user_orders AS
SELECT u.username, o.order_id, o.amount, o.created_at
FROM users u
INNER JOIN orders o ON u.id = o.user_id;

-- 使用视图
SELECT * FROM user_orders WHERE amount > 100;

-- 修改视图
ALTER VIEW user_orders AS
SELECT u.username, u.email, o.order_id
FROM users u
INNER JOIN orders o ON u.id = o.user_id;

-- 删除视图
DROP VIEW IF EXISTS user_orders;

15. 常见问题

15.1 字符集问题

-- 查看字符集
SHOW VARIABLES LIKE 'character%';

-- 设置数据库字符集
ALTER DATABASE mydb CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 设置表字符集
ALTER TABLE users CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 设置列字符集
ALTER TABLE users MODIFY COLUMN username VARCHAR(50) CHARACTER SET utf8mb4;

15.2 连接数问题

-- 查看最大连接数
SHOW VARIABLES LIKE 'max_connections';

-- 设置最大连接数
SET GLOBAL max_connections = 1000;

-- 查看当前连接数
SHOW PROCESSLIST;

-- 杀死连接
KILL CONNECTION_ID;

15.3 锁表问题

-- 查看表锁
SHOW OPEN TABLES WHERE In_use > 0;

-- 查看进程
SHOW PROCESSLIST;

-- 解除锁
UNLOCK TABLES;

总结

MySQL 是一个功能强大且广泛使用的数据库系统。掌握以上内容能够满足大多数应用场景的需求。持续学习和实践是提升 MySQL 技能的最好方法。
