// models/User.ts
export interface User {
  id: number;
  name: string;
  email: string;
}

// utils/Database.ts
export class Database<T extends { id: number }> {
  private data: T[] = [];

  add(item: T): void {
    this.data.push(item);
  }

  findById(id: number): T | undefined {
    return this.data.find(x => x.id === id);
  }

  getAll(): T[] {
    return [...this.data];
  }
}

// main.ts
import { User } from "./models/User";
import { Database } from "./utils/Database";

const userDb = new Database<User>();

userDb.add({ id: 1, name: "Tom", email: "tom@test.com" });
userDb.add({ id: 2, name: "Jerry", email: "jerry@test.com" });

console.log(userDb.getAll());
console.log(userDb.findById(1));
