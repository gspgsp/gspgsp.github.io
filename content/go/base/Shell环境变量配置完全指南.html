Shell зҺҜеўғеҸҳйҮҸй…ҚзҪ®е®Ңе…ЁжҢҮеҚ—пјҲдҝ®жӯЈзүҲпјү

дёҖгҖҒй…ҚзҪ®ж–Үд»¶жҰӮи§Ҳ

ж–Үд»¶дҪңз”ЁеҹҹдҪ•ж—¶еҠ иҪҪйҖӮз”Ё shellдјҳе…Ҳзә§
/etc/profileзі»з»ҹе…ЁеұҖLogin shell ж—¶жүҖжңү shellдҪҺ
/etc/bash.bashrcзі»з»ҹе…ЁеұҖжҜҸдёӘ bash еҗҜеҠЁж—¶bashдёӯ
~/.profileз”ЁжҲ·Login shell ж—¶жүҖжңү shellдёӯ
~/.bashrcз”ЁжҲ·йқһ login bash еҗҜеҠЁж—¶bashй«ҳ
~/.zshrcз”ЁжҲ·йқһ login zsh еҗҜеҠЁж—¶zshй«ҳ

дјҳе…Ҳзә§иҜҙжҳҺпјҡ й«ҳдјҳе…Ҳзә§зҡ„й…ҚзҪ®дјҡиҰҶзӣ–дҪҺдјҳе…Ҳзә§зҡ„й…ҚзҪ®

дәҢгҖҒShell зұ»еһӢиҜҰи§Ј

Login ShellпјҲзҷ»еҪ• shellпјү

е®ҡд№үпјҡ йңҖиҰҒиҫ“е…Ҙз”ЁжҲ·еҗҚе’ҢеҜҶз Ғзҡ„ shell

зңҹе®һеңәжҷҜпјҡ

  • SSH иҝңзЁӢзҷ»еҪ•
  • иҷҡжӢҹжңәеҗҜеҠЁж—¶зҡ„еҲқе§Ӣ TTY е‘Ҫд»ӨиЎҢзҷ»еҪ•пјҲиҫ“е…Ҙз”ЁжҲ·еҗҚеҜҶз Ғпјү
  • bash --login жҳҫејҸеҗҜеҠЁ login shell

еҠ иҪҪйЎәеәҸпјҡ

/etc/profile вҶ’ ~/.profileпјҲжҲ– ~/.bash_profileпјү

Non-login ShellпјҲйқһзҷ»еҪ• shellпјү

е®ҡд№үпјҡ е·Із»Ҹзҷ»еҪ•еҗҺеҶҚжү“ејҖзҡ„ shell

зңҹе®һеңәжҷҜпјҡ

  • еңЁе·Іжү“ејҖзҡ„з»Ҳз«Ҝдёӯиҫ“е…Ҙ bash е‘Ҫд»Ө
  • д»ҺеӣҫеҪўз•Ңйқўжү“ејҖж–°зҡ„з»Ҳз«ҜзӘ—еҸЈ/ж ҮзӯҫйЎө
  • и„ҡжң¬дёӯеҗҜеҠЁзҡ„ bash

еҠ иҪҪйЎәеәҸпјҡ

/etc/bash.bashrc вҶ’ ~/.bashrc

дёүгҖҒиҜҰз»ҶеҠ иҪҪжөҒзЁӢ

еңәжҷҜ AпјҡUbuntu 22 еӣҫеҪўз•ҢйқўејҖжңәпјҲжңҖеёёи§Ғпјү

1. ејҖжңә
2. еҗҜеҠЁеӣҫеҪўз®ЎзҗҶеҷЁпјҲдёҚжҳҜ login shellпјү
3. жҳҫзӨәзҷ»еҪ•з•Ңйқў
4. иҫ“е…Ҙз”ЁжҲ·еҗҚеҜҶз Ғ вҶ’ иҝӣе…ҘжЎҢйқўпјҲдёҚжҳҜ login shellпјү
5. жү“ејҖз»Ҳз«Ҝеә”з”Ё
   вҶ“
   еҠ иҪҪ /etc/bash.bashrcпјҲзі»з»ҹй…ҚзҪ®пјү
   еҠ иҪҪ ~/.bashrcпјҲз”ЁжҲ·й…ҚзҪ®пјү
   вҶ“
   вқҢ /etc/profile е’Ң ~/.profile дёҚдјҡеҠ иҪҪ

з»“и®әпјҡ жЎҢйқўз”ЁжҲ·еңЁжү“ејҖз»Ҳз«Ҝж—¶пјҢеҸӘжңү /etc/bash.bashrc е’Ң ~/.bashrc з”ҹж•Ҳ

еңәжҷҜ BпјҡиҷҡжӢҹжңәжҲ–жңҚеҠЎеҷЁ TTY зҷ»еҪ•

1. ејҖжңә
2. жҳҫзӨә TTY зҷ»еҪ•з•ҢйқўпјҲзңҹжӯЈзҡ„ login shellпјү
3. иҫ“е…Ҙз”ЁжҲ·еҗҚеҜҶз Ғ
   вҶ“
   еҠ иҪҪ /etc/profileпјҲзі»з»ҹе…ЁеұҖй…ҚзҪ®пјү
   еҠ иҪҪ ~/.profileпјҲз”ЁжҲ·й…ҚзҪ®пјү
   еҠ иҪҪ /etc/bash.bashrcпјҲbash зү№е®ҡй…ҚзҪ®пјү
   еҠ иҪҪ ~/.bashrcпјҲз”ЁжҲ· bash й…ҚзҪ®пјү
   вҶ“
   вң… жүҖжңүй…ҚзҪ®йғҪеҠ иҪҪ

з»“и®әпјҡ жңҚеҠЎеҷЁзҷ»еҪ•ж—¶пјҢжүҖжңүй…ҚзҪ®ж–Үд»¶йғҪдјҡжҢүйЎәеәҸеҠ иҪҪ

еңәжҷҜ CпјҡSSH иҝңзЁӢзҷ»еҪ•

1. SSH иҝһжҺҘеҲ°жңҚеҠЎеҷЁ
2. еҗҜеҠЁ login shell
   вҶ“
   еҠ иҪҪ /etc/profile
   еҠ иҪҪ ~/.profile
   еҠ иҪҪ /etc/bash.bashrc
   еҠ иҪҪ ~/.bashrc
   вҶ“
   вң… жүҖжңүй…ҚзҪ®йғҪеҠ иҪҪ

з»“и®әпјҡ SSH зҷ»еҪ•е’Ң TTY зҷ»еҪ•дёҖж ·пјҢйғҪжҳҜ login shell

еӣӣгҖҒзҺҜеўғеҸҳйҮҸз”ҹж•Ҳж—¶жңәеҜ№з…§иЎЁ

еҜ№дәҺеӣҫеҪўз•Ңйқўз”ЁжҲ·пјҲUbuntu 22 жЎҢйқўпјү

й…ҚзҪ®дҪҚзҪ®ејҖжңәеҗҺжү“ејҖз»Ҳз«ҜеҗҺеңЁз»Ҳз«Ҝжү§иЎҢ bashSSH зҷ»еҪ•
/etc/profileвқҢвқҢвқҢвң…
/etc/bash.bashrcвқҢвң…вң…вң…
~/.bashrcвқҢвң…вң…вң…
~/.profileвқҢвқҢвқҢвң…

еҜ№дәҺжңҚеҠЎеҷЁз”ЁжҲ·пјҲTTY жҲ– SSHпјү

й…ҚзҪ®дҪҚзҪ®зҷ»еҪ•ж—¶еҶҚжү“ејҖж–° bashSSH зҷ»еҪ•
/etc/profileвң…вқҢвң…
/etc/bash.bashrcвң…вң…вң…
~/.bashrcвң…вң…вң…
~/.profileвң…вқҢвң…

дә”гҖҒзҺҜеўғеҸҳйҮҸзҡ„дҪңз”ЁзӨәдҫӢ

JAVA_HOME зҡ„дҪңз”Ё

и®ҫзҪ®пјҡ

export JAVA_HOME=/usr/local/jdk-17/jdk-17.0.7

$JAVA_HOME/bin/java
$JAVA_HOME/bin/javac

    # ж”№еҸҳ JAVA_HOME е°ұиғҪеҲҮжҚўзүҲжң¬пјҢдёҚйңҖиҰҒж”№ PATH

    tomcat/bin/startup.sh
# Error: JAVA_HOME is not set  вҶҗ жІЎжңүй…ҚзҪ® JAVA_HOME

GOPATH зҡ„дҪңз”Ё

и®ҫзҪ®пјҡ

export GOPATH=$HOME/go #еҜ№дәҺж–°зүҲжң¬зҡ„go v1.11+ д№ӢеҗҺдёҚйңҖиҰҒиҝҷж ·и®ҫзҪ®, GOPATHй»ҳи®Өе°ұжҳҜиҝҷдёӘ

Go еҰӮдҪ•дҪҝз”Ёе®ғпјҡ

  1. еӯҳеӮЁдҫқиө–еҢ…зҡ„дҪҚзҪ®

    $GOPATH/
в”ңв”Җв”Җ bin/      (go install зҡ„еҸҜжү§иЎҢж–Үд»¶)
в”ңв”Җв”Җ pkg/      (дҫқиө–еҢ…)
в””в”Җв”Җ src/      (йЎ№зӣ®жәҗд»Јз Ғ)

  2. жҹҘжүҫеҜје…Ҙзҡ„еҢ…

    import "github.com/some/package"
// Go дјҡеңЁ $GOPATH/pkg/mod дёӢжҹҘжүҫиҝҷдёӘеҢ…

  3. ж–°ж—§зүҲжң¬еҜ№жҜ”

# ж—§зүҲжң¬ GoпјҲ<1.11пјү
# еҝ…йЎ»и®ҫзҪ® GOPATHпјҢжүҖжңүйЎ№зӣ®йғҪеңЁ $GOPATH/src дёӢ
export GOPATH=$HOME/go  # вқҢ еҝ…йЎ»жүӢеҠЁи®ҫзҪ®

# ж–°зүҲжң¬ GoпјҲ1.11+пјү
# еј•е…Ҙ go.modпјҢйЎ№зӣ®еҸҜд»Ҙж”ҫеңЁд»»дҪ•ең°ж–№
# GOPATH жңүй»ҳи®ӨеҖјпјҢдёҚйңҖиҰҒжүӢеҠЁй…ҚзҪ®
# й»ҳи®ӨеҖјпјҡ$HOME/goпјҲеңЁ Linux/Macпјү
export GOPATH=$HOME/go  # вң… еҸҜйҖүпјҢжңүй»ҳи®ӨеҖј


# вң… ж–°йЎ№зӣ®пјҡдёҚйңҖиҰҒ export GOPATHпјҢзӣҙжҺҘз”Ё go.mod
cd ~/my-project
go mod init github.com/user/my-project #е…ідәҺgo modзҡ„дҪҝз”ЁеҸӮиҖғ: https://go.dev/ref/mod
go build

# вқҢ дёҚз”ЁеҶҚеғҸж—§зүҲжң¬иҝҷж ·пјҡ
# export GOPATH=$HOME/go
# cd $GOPATH/src
# mkdir -p github.com/user/my-project
# cd github.com/user/my-project

PATH зҡ„дҪңз”Ё

и®ҫзҪ®пјҡ

export PATH=$PATH:$JAVA_HOME/bin:$GOPATH/bin

зі»з»ҹеҰӮдҪ•дҪҝз”Ёе®ғпјҡ

  1. еҝ«йҖҹжү§иЎҢе‘Ҫд»ӨпјҢдёҚйңҖиҰҒе®Ңж•ҙи·Ҝеҫ„

    # жңү PATH й…ҚзҪ®
java -version    # зі»з»ҹиҮӘеҠЁеңЁ $PATH дёӯжҹҘжүҫ java

# жІЎжңү PATH й…ҚзҪ®
/usr/local/jdk-17/jdk-17.0.7/bin/java -version  # еҝ…йЎ»еҶҷе®Ңж•ҙи·Ҝеҫ„

  2. е‘Ҫд»ӨжҗңзҙўйЎәеәҸ

    # PATH дёӯзҡ„зӣ®еҪ•жҢүйЎәеәҸжҗңзҙўпјҢжүҫеҲ°з¬¬дёҖдёӘе°ұжү§иЎҢ
# еҰӮжһңжңүеӨҡдёӘ javaпјҢдјҡжү§иЎҢжңҖеүҚйқўзҡ„йӮЈдёӘ
echo $PATH  # жҹҘзңӢжҗңзҙўйЎәеәҸ

е…ӯгҖҒжңҖдҪіе®һи·ө

вң… жҺЁиҚҗпјҡдёӘдәәз”өи„‘з”ЁжҲ·пјҲUbuntu 22 жЎҢйқўпјү

# еҸӘйңҖеңЁ ~/.bashrc дёӯй…ҚзҪ®еҚіеҸҜ
echo 'export JAVA_HOME=/usr/local/jdk-17/jdk-17.0.7' >> ~/.bashrc
echo 'export GOPATH=$HOME/go' >> ~/.bashrc
echo 'export PATH=$PATH:$JAVA_HOME/bin:$GOPATH/bin' >> ~/.bashrc

# йҮҚж–°еҠ иҪҪй…ҚзҪ®
source ~/.bashrc

# йӘҢиҜҒ
echo $JAVA_HOME
java -version
go env GOPATH

дјҳеҠҝпјҡ

  • вң… еңЁжүҖжңүжү“ејҖзҡ„з»Ҳз«ҜдёӯйғҪз”ҹж•Ҳ
  • вң… дёҚйңҖиҰҒйҮҚж–°зҷ»еҪ•
  • вң… дёҚжұЎжҹ“зі»з»ҹй…ҚзҪ®
  • вң… з”ЁжҲ·зӢ¬з«Ӣз»ҙжҠӨ

вҡ пёҸ йҒҝе…ҚпјҡдёҚиҰҒеңЁзі»з»ҹзә§ж–Үд»¶дёӯж”ҫз”ЁжҲ·й…ҚзҪ®

# вқҢ дёҚиҰҒиҝҷж ·еҒҡ
# /etc/bash.bashrc жҲ– /etc/profile
export GOPATH=/home/qinhong/go  # зЎ¬зј–з Ғз”ЁжҲ·еҗҚ

й—®йўҳпјҡ

  • вқҢ е…¶д»–з”ЁжҲ·зҡ„и·Ҝеҫ„й”ҷиҜҜ
  • вқҢ з”ЁжҲ·еҫҲйҡҫиҰҶзӣ–пјҲеҰӮдҪ д№ӢеүҚйҒҮеҲ°зҡ„й—®йўҳпјү
  • вқҢ з»ҙжҠӨеӣ°йҡҫ

в„№пёҸ зі»з»ҹзә§й…ҚзҪ®зҡ„еҗҲзҗҶдҪҝз”Ё

й…ҚзҪ®еҶ…е®№дҪҚзҪ®еҺҹеӣ 
зі»з»ҹ JDKгҖҒPython зӯүејҖеҸ‘е·Ҙе…·/etc/profileжңҚеҠЎеҷЁзҺҜеўғпјҢжүҖжңүз”ЁжҲ·йғҪеҸҜиғҪз”Ё
зі»з»ҹд»ЈзҗҶгҖҒSSL иҜҒд№Ұ/etc/profileзі»з»ҹзә§йңҖжұӮ
bash й»ҳи®ӨиЎҢдёәи®ҫзҪ®/etc/bash.bashrcеҪұе“ҚжүҖжңү bash з”ЁжҲ·
з”ЁжҲ·дёӘдәәи·Ҝеҫ„гҖҒеҜҶй’Ҙ~/.bashrcд»…еҪ“еүҚз”ЁжҲ·пјҢйҡҗз§ҒиҖғиҷ‘

дёғгҖҒдјҳе…Ҳзә§е’ҢиҰҶзӣ–и§„еҲҷ

еҠ иҪҪйЎәеәҸпјҲд»Һе…ҲеҲ°еҗҺпјүпјҡ

Login shell жғ…еҶөдёӢ:
  1. /etc/profile
  2. ~/.profile
  3. /etc/bash.bashrc
  4. ~/.bashrc

Non-login shell жғ…еҶөдёӢ:
  1. /etc/bash.bashrc
  2. ~/.bashrc

иҰҶзӣ–и§„еҲҷпјҡ еҗҺеҠ иҪҪзҡ„й…ҚзҪ®дјҡиҰҶзӣ–е…ҲеҠ иҪҪзҡ„зӣёеҗҢеҸҳйҮҸ

дҫӢеӯҗпјҡ

# /etc/bash.bashrc
export GOPATH=/usr/local/go-project

# ~/.bashrc
export GOPATH=$HOME/go

жңҖз»Ҳз»“жһңпјҡ GOPATH=$HOME/goпјҲз”ЁжҲ·й…ҚзҪ®иҰҶзӣ–зі»з»ҹй…ҚзҪ®пјү

е…«гҖҒжҺ’жҹҘй—®йўҳзҡ„ж–№жі•

жЈҖжҹҘзҺҜеўғеҸҳйҮҸжқҘжәҗ

# жҹҘзңӢеҪ“еүҚзҡ„еҖј
echo $JAVA_HOME

# жҗңзҙўе“Әдәӣж–Үд»¶и®ҫзҪ®дәҶиҝҷдёӘеҸҳйҮҸ
grep -r "JAVA_HOME" /etc/ ~/ 2>/dev/null

# з”Ёи°ғиҜ•жЁЎејҸзңӢеҠ иҪҪиҝҮзЁӢ
bash -x ~/.bashrc 2>&1 | head -20

ејәеҲ¶жё…йҷӨй…ҚзҪ®

# д»Һж–Үд»¶дёӯеҲ йҷӨ
sudo sed -i '/JAVA_HOME/d' /etc/bash.bashrc
sed -i '/JAVA_HOME/d' ~/.bashrc

# еҪ“еүҚ shell дёӯдёҙж—¶жё…йҷӨ
unset JAVA_HOME

# йӘҢиҜҒ
echo $JAVA_HOME  # еә”иҜҘдёәз©ә

жЈҖжҹҘжҳҜеҗҰжҳҜ login shell

# еҰӮжһңиҫ“еҮә -bashпјҢиҜҙжҳҺжҳҜ login shell
# еҰӮжһңиҫ“еҮә bashпјҢиҜҙжҳҺжҳҜйқһ login shell
echo $0

# жҲ–иҖ…
shopt login_shells

д№қгҖҒдҪ зҡ„е…·дҪ“жғ…еҶөжҖ»з»“

й—®йўҳеҺҹеӣ пјҡ

  1. GOPATH й…ҚзҪ®еңЁ /etc/bash.bashrcпјҲзі»з»ҹзә§пјү
  2. е°қиҜ•з”Ё go env -w еңЁз”ЁжҲ·зә§иҰҶзӣ–
