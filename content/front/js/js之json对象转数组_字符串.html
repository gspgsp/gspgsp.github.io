js之json对象转数组_字符串:

1.json对象转字符串
function getStringFromJson(obj) {
  let str = "";
  Object.keys(obj).forEach((v) => {
    str += v + ":" + obj[v] + "\n";
  });
  return str;
}
  
2.json对象转数组
function getStringFromJson(obj) {
  const arr = [];
  Object.keys(obj).forEach((v) => {
    const o = {};
    o[v] = obj[v];
    arr.push(o);
  });
  return arr;
}
