дҪҝз”Ё:
е…ҲйҖҡиҝҮstudioй“ҫжҺҘеҗҜеҠЁзҡ„mongodb(дҪҝз”ЁиҝҷдёӘuri: mongodb://admin:admin123@127.0.0.1:27017/admin)
й“ҫжҺҘжҲҗеҠҹд№ӢеҗҺй»ҳи®ӨжңүеҚ•дёӘж•°жҚ®еә“ admin config local

дёӢйқўжҳҜеҜ№mongoж•°жҚ®еә“зҡ„еҲҶжһҗ:

MongoDB й»ҳи®ӨеҢ…еҗ«д»ҘдёӢдёүдёӘзі»з»ҹж•°жҚ®еә“пјҡ

  1. admin

    • еҠҹиғҪ:
      • admin ж•°жҚ®еә“иў«и®ӨдёәжҳҜ MongoDB зҡ„вҖңе…ЁеұҖвҖқж•°жҚ®еә“пјҢеӯҳеӮЁзі»з»ҹиҢғеӣҙзҡ„й…ҚзҪ®гҖӮ
      • еҰӮжһңеҗ‘ admin ж•°жҚ®еә“ж·»еҠ з”ЁжҲ·е№¶жҺҲдәҲйҖӮеҪ“жқғйҷҗпјҢиҜҘз”ЁжҲ·е°ҶжӢҘжңүжүҖжңүж•°жҚ®еә“зҡ„з®ЎзҗҶжқғйҷҗгҖӮ
    • з”ЁйҖ”:
      • еӯҳеӮЁз”ЁжҲ·жқғйҷҗе’Ңи§’иүІдҝЎжҒҜгҖӮ
      • й…ҚзҪ®зі»з»ҹзә§еҲ«зҡ„еҸӮж•°гҖӮ
    • еёёз”Ёе‘Ҫд»Ө:
      use admin
db.system.users.find()  // жҹҘзңӢз”ЁжҲ·дҝЎжҒҜ

  2. config

    • еҠҹиғҪ:
      • еӯҳеӮЁеҲҶзүҮйӣҶзҫӨзҡ„дҝЎжҒҜпјҲд»…еҪ“еҗҜз”ЁдәҶеҲҶзүҮж—¶жңүз”ЁпјүгҖӮ
      • еҢ…еҗ«дёҺеҲҶзүҮзӣёе…ізҡ„е…ғж•°жҚ®пјҢеҰӮеҲҶзүҮй”®е’ҢеҲҶзүҮеҲҶеёғгҖӮ
    • з”ЁйҖ”:
      • д»…з”ЁдәҺеҲҶзүҮжЁЎејҸзҡ„е…ғж•°жҚ®еӯҳеӮЁгҖӮ
    • жіЁж„Ҹ:
      • еңЁйқһеҲҶзүҮжЁЎејҸдёӢпјҢиҝҷдёӘж•°жҚ®еә“йҖҡеёёдёҚйңҖиҰҒзӣҙжҺҘи®ҝй—®жҲ–дҝ®ж”№гҖӮ

  3. local

    • еҠҹиғҪ:
      • еӯҳеӮЁжң¬ең°иҠӮзӮ№зҡ„зү№е®ҡдҝЎжҒҜгҖӮ
      • ж•°жҚ®дёҚдјҡеңЁеүҜжң¬йӣҶдёӯеҗҢжӯҘгҖӮ
    • з”ЁйҖ”:
      • еӯҳеӮЁ OplogпјҲж“ҚдҪңж—Ҙеҝ—пјүпјҢз”ЁдәҺеүҜжң¬йӣҶзҡ„еӨҚеҲ¶гҖӮ
      • жң¬ең°й…ҚзҪ®жҲ–зҠ¶жҖҒдҝЎжҒҜгҖӮ
    • жіЁж„Ҹ:
      • йҖҡеёёж— йңҖжүӢеҠЁдҝ®ж”№жҲ–еҲ йҷӨжӯӨж•°жҚ®еә“дёӯзҡ„еҶ…е®№гҖӮ
еҰӮдҪ•еҲӣе»әиҮӘе®ҡд№үж•°жҚ®еә“

MongoDB дёҚйңҖиҰҒйў„е…ҲеҲӣе»әж•°жҚ®еә“пјҢжҸ’е…Ҙж•°жҚ®ж—¶дјҡиҮӘеҠЁеҲӣе»әж•°жҚ®еә“гҖӮдҫӢеҰӮпјҡ

  1. еҲҮжҚўеҲ°ж–°ж•°жҚ®еә“пјҲеҰӮ mydatabaseпјүпјҡ

    use mydatabase

  2. жҸ’е…ҘдёҖдёӘйӣҶеҗҲе’Ңж–ҮжЎЈпјҡ

    db.mycollection.insertOne({ name: "Alice", age: 25 })

  3. ж•°жҚ®еә“ mydatabase е’ҢйӣҶеҗҲ mycollection е°Ҷиў«иҮӘеҠЁеҲӣе»әгҖӮ

жҹҘиҜўй»ҳи®Өж•°жҚ®еә“дёӯзҡ„еҶ…е®№

  1. жҹҘзңӢй»ҳи®Өж•°жҚ®еә“еҲ—иЎЁ:

    show dbs

    иҫ“еҮәпјҡ

    admin    0.000GB
config   0.000GB
local    0.000GB

  2. еҲҮжҚўеҲ° admin ж•°жҚ®еә“е№¶жҹҘиҜўз”ЁжҲ·:

    use admin
db.system.users.find()

  3. еҲҮжҚўеҲ° local ж•°жҚ®еә“е№¶жҹҘзңӢ OplogпјҲйҖӮз”ЁдәҺеүҜжң¬йӣҶпјү:

    use local
show collections
db.oplog.rs.find().limit(5).pretty()
жҖ»з»“
  • admin: зі»з»ҹиҢғеӣҙз®ЎзҗҶе’Ңз”ЁжҲ·жқғйҷҗгҖӮ
  • config: еҲҶзүҮйӣҶзҫӨе…ғж•°жҚ®гҖӮ
  • local: жң¬ең°иҠӮзӮ№зҡ„й…ҚзҪ®пјҢдёҚдјҡеңЁйӣҶзҫӨй—ҙеӨҚеҲ¶гҖӮ

еҰӮжһңдҪ жғіеҲӣе»әиҮӘе·ұзҡ„ж•°жҚ®еә“жҲ–иҝӣиЎҢиҝӣдёҖжӯҘж“ҚдҪңпјҢеҸӘйңҖеҲҮжҚўеҲ°дҪ зҡ„зӣ®ж Үж•°жҚ®еә“еҗҚз§°е№¶ејҖе§ӢжҸ’е…Ҙж•°жҚ®еҚіеҸҜпјҒ

иҙҙдёҖдёӘnode.jsй“ҫжҺҘmongoпјҡ

const { MongoClient } = require('mongodb');
  
const uri = "mongodb://admin:admin123@127.0.0.1:27017/admin"; //иҝҷдёӘurlд№ҹеҸҜд»ҘдҪңдёәstudioй“ҫжҺҘmongoзҡ„uri
  
const client = new MongoClient(uri);
  
async function run() {
  try {
    await client.connect();
    console.log("Connected to MongoDB!");
    const db = client.db("test"); // жӣҝжҚўдёәдҪ йңҖиҰҒж“ҚдҪңзҡ„ж•°жҚ®еә“
    const collection = db.collection("example"); // жӣҝжҚўдёәдҪ зҡ„йӣҶеҗҲ
    const docs = await collection.find().toArray();
    console.log(docs);
  } finally {
    await client.close();
  }
}
  
run().catch(console.error);
